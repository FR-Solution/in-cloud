import { CERTIFICATES } from '@site/src/constants/kubernetes/certs'
import { CUSTOM_VALUE } from '@site/src/constants/kubernetes/customValue'

import CodeBlock        from '@theme/CodeBlock'
import dedent           from 'ts-dedent'

<h4>Переменные окружения</h4>
<CodeBlock language="bash">
    {dedent`
        export BASE_K8S_PATH="${CUSTOM_VALUE.kuberneteBaseFolderPath.value}"
        export KUBERNETES_SCHEDULER_CLIENT_CN="${CERTIFICATES.kubernetesSchedulerClient.cname}"
        export KUBERNETES_SCHEDULER_CLIENT_CRT_CONF="${CERTIFICATES.kubernetesSchedulerClient.crtConf}"
        export KUBERNETES_SCHEDULER_CLIENT_CRT_PATH="${CERTIFICATES.kubernetesSchedulerClient.crtPath}"
        export KUBERNETES_SCHEDULER_CLIENT_CSR_PATH="${CERTIFICATES.kubernetesSchedulerClient.csrPath}"
        export KUBERNETES_SCHEDULER_CLIENT_KEY_PATH="${CERTIFICATES.kubernetesSchedulerClient.keyPath}"
        export KUBERNETES_SCHEDULER_CLIENT_KEY_SIZE=${CERTIFICATES.kubernetesSchedulerClient.keySize}
        export KUBERNETES_CA_CRT_PATH="${CERTIFICATES.kubernetesCA.crtPath}"
        export KUBERNETES_CA_KEY_PATH="${CERTIFICATES.kubernetesCA.keyPath}"
    `}
</CodeBlock>

<h4>Рабочая директория</h4>
<CodeBlock language="bash">
    {dedent`
        mkdir -p ${CUSTOM_VALUE.kuberneteBaseFolderPath.value}/pki
        mkdir -p ${CUSTOM_VALUE.kuberneteBaseFolderPath.value}/openssl/csr
    `}
</CodeBlock>

<h4>Конфигурация</h4>
```bash
cat <<EOF > ${KUBERNETES_SCHEDULER_CLIENT_CRT_CONF}
[ req ]
default_bits        = ${KUBERNETES_SCHEDULER_CLIENT_KEY_SIZE}
prompt              = no
default_md          = sha256
distinguished_name  = dn

[ dn ]
CN = ${KUBERNETES_SCHEDULER_CLIENT_CN}

[ v3_ext ]
authorityKeyIdentifier=keyid,issuer:always
basicConstraints=CA:FALSE
keyUsage=keyEncipherment,dataEncipherment
extendedKeyUsage=clientAuth
EOF
```

<h4>Генерация приватного ключа</h4>
```bash
openssl genrsa \
    -out "${KUBERNETES_SCHEDULER_CLIENT_KEY_PATH}" ${KUBERNETES_SCHEDULER_CLIENT_KEY_SIZE}
```

<h4>Генерация CSR</h4>
```bash
openssl req \
    -new \
    -key "${KUBERNETES_SCHEDULER_CLIENT_KEY_PATH}" \
    -out "${KUBERNETES_SCHEDULER_CLIENT_CSR_PATH}" \
    -config "${KUBERNETES_SCHEDULER_CLIENT_CRT_CONF}"
```

<h4>Подпись CSR</h4>
```bash
openssl x509 \
    -req \
    -days 365 \
    -sha256 \
    -outform PEM \
    -CA "${KUBERNETES_CA_CRT_PATH}" \
    -CAkey "${KUBERNETES_CA_KEY_PATH}" \
    -CAcreateserial \
    -in "${KUBERNETES_SCHEDULER_CLIENT_CSR_PATH}" \
    -out "${KUBERNETES_SCHEDULER_CLIENT_CRT_PATH}" \
    -extensions v3_ext \
    -extfile "${KUBERNETES_SCHEDULER_CLIENT_CRT_CONF}"
```
