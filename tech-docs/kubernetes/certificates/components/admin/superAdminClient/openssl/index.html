Перейти к основному содержимому
B-CloudТехническая документация

openssl

Переменные окружения

export BASE_K8S_PATH="/etc/kubernetes"
export KUBERNETES_SUPER_ADMIN_CLIENT_CN="kubernetes-super-admin"
export KUBERNETES_SUPER_ADMIN_CLIENT_CRT_CONF="/etc/kubernetes/openssl/super-admin.conf"
export KUBERNETES_SUPER_ADMIN_CLIENT_CRT_PATH="/etc/kubernetes/kubeconfig/super-admin.crt"
export KUBERNETES_SUPER_ADMIN_CLIENT_CSR_PATH="/etc/kubernetes/openssl/csr/super-admin.csr"
export KUBERNETES_SUPER_ADMIN_CLIENT_KEY_PATH="/etc/kubernetes/kubeconfig/super-admin.key"
export KUBERNETES_SUPER_ADMIN_CLIENT_KEY_SIZE=2048
export KUBERNETES_CA_CRT_PATH="/etc/kubernetes/pki/ca.crt"
export KUBERNETES_CA_KEY_PATH="/etc/kubernetes/pki/ca.key"

Рабочая директория

mkdir -p ${BASE_K8S_PATH}/pki
mkdir -p ${BASE_K8S_PATH}/openssl/csr
mkdir -p ${BASE_K8S_PATH}/kubeconfig

Конфигурация

cat <<EOF > ${KUBERNETES_SUPER_ADMIN_CLIENT_CRT_CONF}
[ req ]
default_bits        = ${KUBERNETES_SUPER_ADMIN_CLIENT_KEY_SIZE}
prompt              = no
default_md          = sha256
distinguished_name  = dn

[ dn ]
CN = ${KUBERNETES_SUPER_ADMIN_CLIENT_CN}
O  = system:masters

[ v3_ext ]
authorityKeyIdentifier=keyid,issuer:always
basicConstraints=CA:FALSE
keyUsage=keyEncipherment,dataEncipherment
extendedKeyUsage=clientAuth
EOF

Генерация приватного ключа

openssl genrsa \
    -out "${KUBERNETES_SUPER_ADMIN_CLIENT_KEY_PATH}" ${KUBERNETES_SUPER_ADMIN_CLIENT_KEY_SIZE}

Генерация CSR

openssl req \
    -new \
    -key    "${KUBERNETES_SUPER_ADMIN_CLIENT_KEY_PATH}" \
    -out    "${KUBERNETES_SUPER_ADMIN_CLIENT_CSR_PATH}" \
    -config "${KUBERNETES_SUPER_ADMIN_CLIENT_CRT_CONF}"

Подпись CSR

openssl x509 \
    -req \
    -days 365 \
    -sha256 \
    -CA "${KUBERNETES_CA_CRT_PATH}" \
    -CAkey "${KUBERNETES_CA_KEY_PATH}" \
    -CAcreateserial \
    -in "${KUBERNETES_SUPER_ADMIN_CLIENT_CSR_PATH}" \
    -out "${KUBERNETES_SUPER_ADMIN_CLIENT_CRT_PATH}" \
    -extensions v3_ext \
    -extfile "${KUBERNETES_SUPER_ADMIN_CLIENT_CRT_CONF}"
