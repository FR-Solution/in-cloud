Настройка инструмента взаимодействия с ETCD​

cat <<'EOF' >> ~/.bashrc
export BASE_K8S_PATH="/etc/kubernetes"
export ETCDCTL_CERT="${BASE_K8S_PATH}/pki/etcd/peer.crt"
export ETCDCTL_KEY="${BASE_K8S_PATH}/pki/etcd/peer.key"
export ETCDCTL_CACERT="${BASE_K8S_PATH}/pki/etcd/ca.crt"
export ETCD_SERVER_PORT="2379"

etcd_endpoints () {
export ENDPOINTS=$(
etcdctl --endpoints=127.0.0.1:${ETCD_SERVER_PORT} member list |
awk '{print $5}' |
sed "s/,//" |
sed "s/ /,/g"
)
}

estat () {
etcd_endpoints
etcdctl --write-out=table --endpoints=$ENDPOINTS endpoint status
}
EOF

source ~/.bashrc

примечание

Если вы когда-либо заглядывали в документацию от Келси Хайтауэра, то, возможно, заметили, что в конфигурационных файлах часто используются IP-адреса узлов. Этот метод, безусловно, работает, но есть более удобный способ. Чтобы упростить обслуживание и сделать дальнейшую шаблонизацию более гибкой, лучше использовать заранее известные FQDN-маски, как я уже упоминал для мастеров выше master-${INDEX}.${CLUSTER_NAME}.${BASE_DOMAIN}.
