configFiles

Переменные окружния

export BASE_K8S_PATH="/etc/kubernetes"
export BASE_KUBELET_PATH="/var/lib/kubelet"
export KUBERNETES_CA_CRT_PATH="${BASE_K8S_PATH}/pki/ca.crt"
export SERVICE_DNS="29.64.0.10"
export CLUSTER_NAME="example"
export BASE_DOMAIN="example.com"
export CLUSTER_DOMAIN="${CLUSTER_NAME}.${BASE_DOMAIN}"
export KUBELET_CURRENT_SERVER_CRT_PATH="${BASE_KUBELET_PATH}/pki/kubelet-server-current.pem"
export KUBELET_HEALTHZ_PORT=10248
