5.1.2.2.3.1. Переменные окружения PKI​

export CLUSTER_NAME="example"
export BASE_DOMAIN="example.com"
export CLUSTER_DOMAIN="${CLUSTER_NAME}.${BASE_DOMAIN}"
export FULL_HOST_NAME="${HOST_NAME}.${CLUSTER_DOMAIN}"
export BASE_K8S_PATH="/etc/kubernetes"
export ETCD_PEER_CN=""
export ETCD_PEER_CRT_CONF="${BASE_K8S_PATH}/openssl/etcd-peer.conf"
export ETCD_PEER_CRT_PATH="${BASE_K8S_PATH}/pki/etcd/peer.crt"
export ETCD_PEER_CSR_PATH="${BASE_K8S_PATH}/openssl/csr/etcd-peer.csr"
export ETCD_PEER_KEY_PATH="${BASE_K8S_PATH}/pki/etcd/peer.key"
export ETCD_PEER_KEY_SIZE=2048
export ETCD_CA_CRT_PATH="${BASE_K8S_PATH}/pki/etcd/ca.crt"
export ETCD_CA_KEY_PATH="${BASE_K8S_PATH}/pki/etcd/ca.key"
export MACHINE_LOCAL_ADDRESS="$(ip -4 addr show scope global | awk '/inet/ {print $2; exit}' | cut -d/ -f1)"

5.1.2.2.3.2. Рабочая директория для PKI​

mkdir -p ${BASE_K8S_PATH}/pki
mkdir -p ${BASE_K8S_PATH}/openssl/csr

5.1.2.2.3.3. Конфигурация PKI​

cat <<EOF > ${ETCD_PEER_CRT_CONF}
[ req ]
default_bits        = ${ETCD_PEER_KEY_SIZE}
prompt              = no
default_md          = sha256
distinguished_name  = dn
req_extensions      = req_ext

[ req_ext ]
subjectAltName = @alt_names

[ alt_names ]
DNS.1 = localhost
DNS.2 = ${HOST_NAME}
DNS.3 = ${FULL_HOST_NAME}
IP.1  = 127.0.0.1
IP.2  = 0:0:0:0:0:0:0:1
IP.3  = ${MACHINE_LOCAL_ADDRESS}

[ dn ]
CN = "${FULL_HOST_NAME}"

[ v3_ext ]
authorityKeyIdentifier=keyid,issuer:always
basicConstraints=CA:FALSE
keyUsage=keyEncipherment,dataEncipherment
extendedKeyUsage=serverAuth,clientAuth
subjectAltName=@alt_names
EOF

5.1.2.2.3.4. Генерация приватного ключа​

openssl genrsa \
    -out "${ETCD_PEER_KEY_PATH}" ${ETCD_PEER_KEY_SIZE}

5.1.2.2.3.5. Генерация CSR​

openssl req \
    -new \
    -key "${ETCD_PEER_KEY_PATH}" \
    -out "${ETCD_PEER_CSR_PATH}" \
    -config "${ETCD_PEER_CRT_CONF}"

5.1.2.2.3.6. Подпись CSR​

openssl x509 \
    -req \
    -days 365 \
    -sha256 \
    -CA "${ETCD_CA_CRT_PATH}" \
    -CAkey "${ETCD_CA_KEY_PATH}" \
    -CAcreateserial \
    -in "${ETCD_PEER_CSR_PATH}" \
    -out "${ETCD_PEER_CRT_PATH}" \
    -extensions v3_ext \
    -extfile "${ETCD_PEER_CRT_CONF}"
